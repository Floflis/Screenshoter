const { app, BrowserWindow, ipcMain } = require('electron');
const path = require('path');
const screenshot = require('electron-screenshot');

const exec = require('child_process').exec;

let mainWindow;

function createWindow() {
  mainWindow = new BrowserWindow({
    width: 800,
    height: 600,
    webPreferences: {
      nodeIntegration: true,
      contextIsolation: false,
      enableRemoteModule: true
    }
  });

  mainWindow.loadURL(`file://${path.join(__dirname, '../build/index.html')}`);

  mainWindow.on('closed', () => {
    mainWindow = null;
  });
}

app.on('ready', createWindow);

app.on('window-all-closed', () => {
  if (process.platform !== 'darwin') {
    app.quit();
  }
});

app.on('activate', () => {
  if (mainWindow === null) {
    createWindow();
  }
});

ipcMain.handle('screenshot', async (event) => {
  const imgPath = await screenshot({
    filename: 'screenshot.png'
  });

  return imgPath;
});

ipcMain.handle('scrollScreenshot', async (event, scrollTop) => {
  const imgPath = await screenshot({
    filename: 'screenshot.png',
    x: 0,
    y: scrollTop
  });

  return imgPath;
});

ipcMain.handle('scrollWindow', async (event, scrollTop) => {
  return new Promise((resolve, reject) => {
    const scrollScriptPath = path.join(__dirname, './scroll.sh');
    const cmd = `bash ${scrollScriptPath} ${scrollTop}`;
    exec(cmd, (error, stdout, stderr) => {
      if (error) {
        console.error(`exec error: ${error}`);
        reject(error);
        return;
      }
      resolve(stdout.trim());
    });
  });
});

You will need to install the required dependencies by running npm install electron-screenshot child_process in the app directory.

This code sets up a basic Electron app with a window containing a screenshot button and an arrow-down button. Clicking the screenshot button will capture a screenshot of the current window and return the image path to the renderer process. Clicking the arrow-down button will trigger a scroll event that calls an external shell script (scroll.sh) to scroll the window by the height of the screen, take a new screenshot, and return the image path to the renderer process. The shell script is responsible for simulating the scroll event and relies on Linux command-line tools like xdotool and import to take the screenshot and concatenate them.

Sure! Here's an example scroll.sh script that scrolls down the current window by a specified number of pixels and takes a screenshot:

bash
#!/bin/bash

# Get current window ID
WINDOW_ID=$(xdotool getactivewindow)

# Get current window dimensions
WINDOW_WIDTH=$(xwininfo -id $WINDOW_ID | grep "Width:" | awk '{print $2}')
WINDOW_HEIGHT=$(xwininfo -id $WINDOW_ID | grep "Height:" | awk '{print $2}')

# Set scroll amount to screen height
SCROLL_AMOUNT=$WINDOW_HEIGHT

# Scroll down
xdotool mousemove_relative 0 $WINDOW_HEIGHT
xdotool click 1

# Wait for the scroll animation to finish
sleep 1

# Take a screenshot of the current window
electron-screenshot -o screenshot.png $WINDOW_ID

# Move the mouse back to the top of the window
xdotool mousemove_relative 0 -$WINDOW_HEIGHT

exit 0

This script uses xdotool to interact with the window manager and simulate mouse clicks to scroll down the window. It also uses electron-screenshot to take a screenshot of the current window.

To scroll down the window, the script simulates a mouse click at the bottom of the window and waits for the scroll animation to finish (in this case, 1 second).

After taking the screenshot, the mouse is moved back to the top of the window to prepare for the next scroll. The WINDOW_ID variable is set to the ID of the current window, which can be passed as an argument to electron-screenshot to take a screenshot of the current window.

One Linux alternative for Microsoft's Image Composite Editor is Hugin. It is a free and open-source panorama photo stitching software available for Linux, macOS, and Windows. It has a graphical user interface, but also provides a command-line interface for automation.

To stitch multiple screenshots into one seamless image without repeating objects that appear in the images, you can use the align_image_stack and enblend command-line tools that come with Hugin.

Here's an example shell script that stitches multiple screenshots into one seamless image using Hugin:

bash
#!/bin/bash

# Specify the path to the screenshots directory
screenshots_dir="/path/to/screenshots/dir"

# Specify the output file name
output_file="output.jpg"

# Get a list of all the screenshot image files in the directory
screenshot_files=($screenshots_dir/*.png)

# Use align_image_stack to align the images
align_image_stack -a aligned_ "${screenshot_files[@]}"

# Use enblend to blend the aligned images into a seamless panorama
enblend -o "$output_file" aligned_*.tif

# Remove the temporary aligned images
rm aligned_*.tif

This script uses align_image_stack to align the screenshot images and enblend to blend them into a seamless panorama. The output is saved as a JPEG file with the specified file name.
